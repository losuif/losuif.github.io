### <span style="color:#7373c9">javascript</span>

```javascript
let tagStack = "";
// 전역변수로 만들면 내용이 제거되지 않고 유지됨
// > 결과적으로 누적

let opt = 0;

function fnTagAdd(){

    opt++;

    let tagContent = "<div class='opt'>옵션" + opt 
                    + "<span class='price'>3000</span></div>";

    if(opt <= 4){
        tagStack += tagContent;
        let res = document.querySelector("#res");
        res.innerHTML = tagStack;
    }
}
```
