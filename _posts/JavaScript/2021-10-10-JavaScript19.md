### <span style="color:#87a333">JavaScript</span>

```javascript
$(function(){

    // 모달 레이어 팝업 열기
    $("div#noticeLayerPop a").click(function(){
        $("div#layerBG").css({"display": "block"});
    
    });


    // 모달 레이어 팝업 닫기
    $("div#closeBtnArea").click(function(){
        $("div#layerBG").css({"display": "none"});
    });

});
```
